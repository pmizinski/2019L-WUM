def get_all_patches(dir = '../data/photos'):
    jpg_files = [f for f in glob.glob(dir + "/*.jpg", recursive=False)]
    #print(jpg_files)
    patches_all = np.empty((0,128))
    sift = cv2.xfeatures2d.SIFT_create()
    for fname in jpg_files:
        img = cv2.imread(fname)
        gray_img = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
        key_points = sift.detect(gray_img, None)
        key_points, img_patches = sift.compute(gray_img, key_points)
        patches_all = np.append(patches_all, img_patches, axis=0)
    return patches_all

def get_centroids(patches_all, n_features = 248):
    kmeans = KMeans(n_clusters=n_features).fit(patches_all)
    return kmeans.cluster_centers_

def histogram(fname, centroids, n_neighbours = 5):
    n_features = centroids.shape[0]
    histogram = [0]*n_features
    labels = [i for i in range(0, n_features)]
    
    sift = cv2.xfeatures2d.SIFT_create()
    img = cv2.imread(fname)
    gray_img = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    key_points = sift.detect(gray_img, None)
    key_points, img_patches = sift.compute(gray_img, key_points)
    
    n = img_patches.shape[0]
    knn = KNeighborsClassifier(n_neighbours)
    knn.fit(centroids, labels)
    pred_labels = knn.predict(img_patches)
    
    for i in range(0, n):
        histogram[pred_labels[i]] += 1
        
    return histogram

def imgs_to_features(centroids, dir = '../data/photos', n_neighbours = 5):
    jpg_files = [f for f in glob.glob(dir + "/*.jpg", recursive=False)]
    
    sift = cv2.xfeatures2d.SIFT_create()
    
    n_features = centroids.shape[0]
    
    features = np.empty((0, n_features))
    actors = []
    emotions = []
    
    for fname in jpg_files:
        hist = np.asarray(histogram(fname, centroids, n_neighbours))
        hist = hist.reshape((1, n_features))
        
        n = len(fname)
        features = np.append(features, hist, axis=0)
        actors.append(fname[n-(8-1)])
        emotions.append(int(fname[(n-(7-1)):(n-(5-1))]))
    
    df_labels = pd.DataFrame({'actor': actors, 'emotion':emotions})
        
    return features, df_labels

def non_zero_columns(arr):
    each_col = np.count_nonzero(arr, axis = 0)
    return np.count_nonzero(each_col)/each_col.size
